    95→-- Passive Lantern Light for MK3 Ballistic Bot
    96→-- Two separate lights: one fixed (small, steady, yellow) and one pulse (large, periodic, yellow)
    97→-- Both only active at night (auto-toggle)
    98→
    99→-- Lightning yellow colour
   100→local LIGHT_R, LIGHT_G, LIGHT_B = 1, 1, 0.3
   101→
   102→-- Fixed light: small steady glow, always on at night
   103→local FIX_RADIUS = 1.5
   104→local FIX_INTENSITY = 0.8
   105→local FIX_FALLOFF = 0.5
   106→
   107→-- Pulse light: large flash, ~same size as a max campfire
   108→local PULSE_RADIUS = 5.5
   109→local PULSE_INTENSITY = 0.85
   110→local PULSE_FALLOFF = 0.5
   111→local PULSE_INTERVAL = 1.5    -- seconds between pulses
   112→local PULSE_DURATION = 0.35   -- how long each pulse flash lasts
   113→
   114→-- Registered prefab for the MK3 pulse light (DST requires registered prefabs
   115→-- for network replication; CreateEntity() without a prefab crashes with
   116→-- "AllocReplica Invalid Prefab").
   117→local function ballistic_pulse_light_fn()
   118→    local inst = CreateEntity()
   119→    inst.entity:AddTransform()
   120→    inst.entity:AddNetwork()
   121→    inst.entity:AddLight()
   122→    inst.Light:Enable(false)
   123→    inst.Light:SetRadius(0)
   124→    inst.Light:SetIntensity(PULSE_INTENSITY)
   125→    inst.Light:SetFalloff(PULSE_FALLOFF)
   126→    inst.Light:SetColour(LIGHT_R, LIGHT_G, LIGHT_B)
   127→    inst.persists = false
   128→    return inst
   129→end

   131→local function StartLightOrb(inst)
   132→    if inst._lightorb_active then return end
   133→    if inst.components.fueled:IsEmpty() then return end
   134→    if not TheWorld.state.isnight then return end
   135→    inst._lightorb_active = true
   136→    inst:AddTag("lantern")
   137→
   138→    -- FIXED LIGHT: set once, never changes while active
   139→    inst.Light:SetRadius(FIX_RADIUS)
   140→    inst.Light:SetIntensity(FIX_INTENSITY)
   141→    inst.Light:SetFalloff(FIX_FALLOFF)
   142→    inst.Light:SetColour(LIGHT_R, LIGHT_G, LIGHT_B)
   143→    inst.Light:Enable(true)
   144→
   145→    -- PULSE LIGHT: re-spawn if missing (e.g. after load or if removed)
   146→    if not inst._pulse_light or not inst._pulse_light:IsValid() then
   147→        inst._pulse_light = SpawnPrefab("ballistic_pulse_light")
   148→        if inst._pulse_light then
   149→            inst._pulse_light.entity:SetParent(inst.entity)
   150→            inst._pulse_light.Transform:SetPosition(0, 1, 0)
   151→        else
   152→            return  -- Cannot spawn pulse light
   153→        end
   154→    end
   155→
   156→    -- Periodic pulse: flash big, then dim, repeat
   157→    inst._lightorb_tick = inst:DoPeriodicTask(PULSE_INTERVAL, function()
   158→        if not inst._lightorb_active then return end
   159→        if inst.components.fueled:IsEmpty() then
   160→            StopLightOrb(inst)
   161→            return
   162→        end
   163→
   164→        -- Flash the pulse light ON
   165→        if inst._pulse_light and inst._pulse_light:IsValid() then
   166→            inst._pulse_light.Light:SetRadius(PULSE_RADIUS)
   167→            inst._pulse_light.Light:Enable(true)
   168→        end
   169→
   170→        -- Spawn spark FX
   171→        if inst._lightorb_fx and inst._lightorb_fx:IsValid() then
   172→            inst._lightorb_fx:Remove()
   173→        end
   174→        local newfx = SpawnPrefab("wx78_big_spark")
   175→        if newfx then
   176→            newfx.entity:SetParent(inst.entity)
   177→            newfx.Transform:SetPosition(0, 1, 0)
   178→            inst._lightorb_fx = newfx
   179→        end
   180→
   181→        -- Turn pulse OFF after short duration
   182→        inst._pulse_dim_task = inst:DoTaskInTime(PULSE_DURATION, function()
   183→            if inst._pulse_light and inst._pulse_light:IsValid() then
   184→                inst._pulse_light.Light:SetRadius(0)
   185→                inst._pulse_light.Light:Enable(false)
   186→            end
   187→            if inst._lightorb_fx and inst._lightorb_fx:IsValid() then
   188→                inst._lightorb_fx:Remove()
   189→                inst._lightorb_fx = nil
   190→            end
   191→        end)
   192→    end)
   193→end
   194→
   195→local function StopLightOrb(inst)
   196→    inst._lightorb_active = false
   197→
   198→    -- Remove pulse light entity (spawned via SpawnPrefab, so safe to Remove)
   199→    if inst._pulse_dim_task then
   200→        inst._pulse_dim_task:Cancel()
   201→        inst._pulse_dim_task = nil
   202→    end
   203→    if inst._pulse_light and inst._pulse_light:IsValid() then
   204→        inst._pulse_light:Remove()
   205→        inst._pulse_light = nil
   206→    end
   207→
   208→    -- Remove FX
   209→    if inst._lightorb_fx and inst._lightorb_fx:IsValid() then
   210→        inst._lightorb_fx:Remove()
   211→    end
   212→    inst._lightorb_fx = nil
   213→
   214→    -- Disable fixed light
   215→    if inst.Light then
   216→        inst.Light:Enable(false)
   217→    end
   218→    inst:RemoveTag("lantern")
   219→
   220→    -- Stop tick
   221→    if inst._lightorb_tick ~= nil then
   222→        inst._lightorb_tick:Cancel()
   223→        inst._lightorb_tick = nil
   224→    end
   225→end

   227→local function maketurret(inst, pt, charge)
   228→    local is_mk3 = inst.was_mk3 or (inst.upgradelevel_mk3 ~= nil and inst.upgradelevel_mk3 >= 150)
   229→    local is_mk2 = is_mk3 or inst.was_mk2 or (inst.upgradelevel ~= nil and inst.upgradelevel >= 100)
   230→    local prefab = is_mk3 and "williamballistic3" or (is_mk2 and "williamballistic2" or "williamballistic")
   231→    local bot = SpawnPrefab(prefab)
   232→    if bot ~= nil then
   233→        bot.Physics:SetCollides(false)
   234→        bot.Physics:Teleport(pt.x, 0, pt.z)
   235→        bot.Physics:SetCollides(true)
   236→        bot.sg:GoToState("revived", charge)
   237→        bot.components.health:SetCurrentHealth(inst.components.health.currenthealth)
   238→        bot.components.fueled.currentfuel = inst.components.fueled.currentfuel
   239→        bot.level = inst.level
   240→        bot.upgradelevel = inst.upgradelevel or 0
   241→        if is_mk2 then
   242→            bot.upgradelevel_mk3 = inst.upgradelevel_mk3 or 0
   243→        end
   244→        bot:PushEvent("levelup")
   245→        inst:Remove()
   246→    end

   283→local function retargetfn(inst)
   284→    local playertargets = {}
   285→    for i, v in ipairs(AllPlayers) do
   286→        if v.components.combat.target ~= nil then
   287→            playertargets[v.components.combat.target] = true
   288→        end
   289→    end
   290→
   291→    return FindEntity(inst, PLACER_SCALE*10,
   292→        function(guy)
   293→            if not inst.components.combat:CanTarget(guy) then return false end
   294→            -- Target enemies already fighting the player/minions
   295→            if playertargets[guy] then return true end
   296→            -- Also target any hostile creature (monster) in range
   297→            if guy:HasTag("monster") and not guy:HasTag("player") and not guy:HasTag("companion") then
   298→                return true
   299→            end
   300→            -- Target creatures attacking players or willminion
   301→            if guy.components.combat.target ~= nil
   302→                and (guy.components.combat.target:HasTag("player") or guy.components.combat.target:HasTag("willminion")) then
   303→                return true
   304→            end
   305→            return false
   306→        end,
   307→        { "_combat" }, --see entityreplica.lua
   308→        { "INLIMBO", "player" }
   309→    )
   310→end
