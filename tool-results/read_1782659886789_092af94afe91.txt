    25‚Üí    25‚Üí-- v2.0.43: Affinity Ramp System + Hit/Blast FX for Sentry MK3
    26‚Üí    26‚Üí----------------------------------------------------------------
    27‚Üí    27‚Üí-- Ramp: each hit on an aligned mob (shadow_aligned for celestial,
    28‚Üí    28‚Üí-- lunar_aligned for shadow) adds +10% base damage as bonus, stacking
    29‚Üí    29‚Üí-- up to 6 times (+60%). The ramp resets if the sentry doesn't attack
    30‚Üí    30‚Üí-- for 3 seconds OR gets hit (mirror of shadow_battleaxe ramp design).
    31‚Üí    31‚Üí-- The x2-Damage proc (10% chance) doubles the CURRENT total damage
    32‚Üí    32‚Üí-- (base + ramp bonus), not just the base ‚Äî so a fully-ramped x2 proc
    33‚Üí    33‚Üí-- is a big burst.
    34‚Üí    34‚Üí--
    35‚Üí    35‚Üí-- FX: bullet hits spawn a brightsmithy-style sparkle (celestial) or
    36‚Üí    36‚Üí-- shadowcraft-style dark spike (shadow). Rocket impacts spawn a
    37‚Üí    37‚Üí-- bomb_lunarplant explosion FX (altered ‚Äî tinted black for shadow).
    38‚Üí    38‚Üí-- All spawns are pcall-guarded with fallbacks so a missing DST prefab
    39‚Üí    39‚Üí-- never crashes the server.
    40‚Üí    40‚Üí
    41‚Üí    41‚Üílocal AFF_RAMP_MAX        = 4      -- v2.0.50: was 6. Max 4 stacks (+40% total)
    42‚Üí    42‚Üí                                    -- ‚Äî 6 stacks (+60%) was OP combined with
    43‚Üí    43‚Üí                                    -- the x2-Damage proc doubling the total.
    44‚Üí    44‚Üílocal AFF_RAMP_PCT        = 0.10   -- +10% base damage per stack
    45‚Üí    45‚Üílocal AFF_RAMP_RESET_TIME = 3      -- seconds without a hit before reset
    46‚Üí    46‚Üí-- v2.0.61: x2-Damage reworked. Base proc is now 4% (down from 10%),
    47‚Üí    47‚Üí-- scaling +2.5% per MK3 sentry the owner has clustered nearby (cached count
    48‚Üí    48‚Üí-- refreshed every few seconds in the periodic task). A 4-second per-target
    49‚Üí    49‚Üí-- cooldown prevents melting a single enemy with back-to-back procs.
    50‚Üí    50‚Üí-- v2.0.65: per-sentry bonus 2.5% -> 3%. One MK3 sentry = 7%, two = 10%.
    51‚Üí    51‚Üílocal AFF_X2_BASE_CHANCE    = 0.04   -- 4% base
    52‚Üí    52‚Üílocal AFF_X2_PER_MK3_BONUS  = 0.03   -- +3% per MK3 sentry (incl. self)

    62‚Üí    62‚Üílocal function ResetAffRamp(inst)
    63‚Üí    63‚Üí    if not inst or not inst:IsValid() then return end
    64‚Üí    64‚Üí    inst._aff_ramp_stacks = 0
    65‚Üí    65‚Üí    if inst._aff_ramp_reset_task then
    66‚Üí    66‚Üí        inst._aff_ramp_reset_task:Cancel()
    67‚Üí    67‚Üí        inst._aff_ramp_reset_task = nil
    68‚Üí    68‚Üí    end
    69‚Üí    69‚Üíend

    71‚Üí    71‚Üílocal function BumpAffRamp(inst)
    72‚Üí    72‚Üí    if not inst or not inst:IsValid() then return end
    73‚Üí    73‚Üí    if inst._aff_ramp_stacks == nil then inst._aff_ramp_stacks = 0 end
    74‚Üí    74‚Üí    if inst._aff_ramp_stacks < AFF_RAMP_MAX then
    75‚Üí    75‚Üí        inst._aff_ramp_stacks = inst._aff_ramp_stacks + 1
    76‚Üí    76‚Üí    end
    77‚Üí    77‚Üí    -- (Re)schedule the 3s no-hit decay timer.
    78‚Üí    78‚Üí    if inst._aff_ramp_reset_task then
    79‚Üí    79‚Üí        inst._aff_ramp_reset_task:Cancel()
    80‚Üí    80‚Üí    end
    81‚Üí    81‚Üí    inst._aff_ramp_reset_task = inst:DoTaskInTime(AFF_RAMP_RESET_TIME, function()
    82‚Üí    82‚Üí        ResetAffRamp(inst)
    83‚Üí    83‚Üí    end)
    84‚Üí    84‚Üíend

   771‚Üí   771‚Üí    local function SetupMK3Affinity(inst)
   772‚Üí   772‚Üí        if inst._affinity_setup_done then return end
   773‚Üí   773‚Üí        inst._affinity_setup_done = true

   776‚Üí   776‚Üí
   777‚Üí   777‚Üí        -- v2.0.43: Ramp state. The periodic task below only SETS the affinity
   778‚Üí   778‚Üí        -- type flag + x2 flag ‚Äî the actual damage logic lives in a SINGLE
   779‚Üí   779‚Üí        -- onhitotherfn set once here (no more swapping onhitotherfn every
   780‚Üí   780‚Üí        -- tick, which was fragile and reset closures constantly).
   781‚Üí   781‚Üí        inst._aff_ramp_stacks = 0

   785‚Üí   785‚Üí
   786‚Üí   786‚Üí        -- Expose ramp helpers on the inst so esentry_rocket can call them
   787‚Üí   787‚Üí        -- (the rocket reads sentry._aff_type / calls sentry.BumpAffRamp etc.
   788‚Üí   788‚Üí        -- to apply the ramp bonus + x2 on rocket direct hits).
   789‚Üí   789‚Üí        inst.ResetAffRamp   = ResetAffRamp
   790‚Üí   790‚Üí        inst.BumpAffRamp    = BumpAffRamp
   791‚Üí   791‚Üí        inst.GetAffRampBonus = GetAffRampBonus
   792‚Üí   792‚Üí
   793‚Üí   793‚Üí        -- Unified onhitotherfn: handles ramp + x2 + FX for bullets.
   794‚Üí   794‚Üí        -- Fires on every bullet hit (the sentry's weapon projectile).
   795‚Üí   795‚Üí        inst.components.combat.onhitotherfn = function(i, other, dmg)
   796‚Üí   796‚Üí            if not other or not other:IsValid() then return end
   797‚Üí   797‚Üí            -- v2.0.51: adaptive affinity per-target. For single-affinity owners
   798‚Üí   798‚Üí            -- this resolves to _aff_type as before. For dual-affinity owners
   799‚Üí   799‚Üí            -- (rare) it picks the affinity matching THIS target's faction, so
   800‚Üí   800‚Üí            -- the ramp applies to whichever aligned mob is being hit.
   801‚Üí   801‚Üí            local aff = i._aff_type
   802‚Üí   802‚Üí            if i._owner_celestial and other:HasTag("shadow_aligned") then
   803‚Üí   803‚Üí                aff = "celestial"

   808‚Üí   808‚Üí            local is_aligned = aligned_tag and other:HasTag(aligned_tag) or false
   809‚Üí   809‚Üí
   810‚Üí   810‚Üí            -- Ramp: bump + apply bonus damage ONLY on aligned hits.

   817‚Üí   817‚Üí                end

   819‚Üí   819‚Üí                if WagstaffAffFX and WagstaffAffFX.HitFX then
   820‚Üí   820‚Üí                    WagstaffAffFX.HitFX(other, aff)
   821‚Üí   821‚Üí                end
   822‚Üí   822‚Üí            end
   823‚Üí   823‚Üí

   825‚Üí   825‚Üí            -- with a 4s per-target cooldown so a single enemy can't be melted
   826‚Üí   826‚Üí            -- by back-to-back procs. Doubles the CURRENT total (base dmg + ramp).
   827‚Üí   827‚Üí            if i._aff_x2_damage then
   828‚Üí   828‚Üí                local cd_until = other._william_x2_cd or 0

   834‚Üí   834‚Üí                            other.components.health:DoDelta(-total, false, "x2_damage")
   835‚Üí   835‚Üí                        end
   836‚Üí   836‚Üí                        other._william_x2_cd = GetTime() + AFF_X2_TARGET_CD
   837‚Üí   837‚Üí                    end
   838‚Üí   838‚Üí                end
   839‚Üí   839‚Üí            end
   840‚Üí   840‚Üí        end
   841‚Üí   841‚Üí
   842‚Üí   842‚Üí        -- Reset the ramp when the sentry gets hit (mirror shadow_battleaxe design:
   843‚Üí   843‚Üí        -- ramp rewards sustained offense, getting hit punishes it).
   844‚Üí   844‚Üí        inst:ListenForEvent("attacked", function(i)
   845‚Üí   845‚Üí            ResetAffRamp(i)
   846‚Üí   846‚Üí        end)
   847‚Üí   847‚Üí
   848‚Üí   848‚Üí        -- Periodic task: refresh the affinity type + x2 flag from the owner
   849‚Üí   849‚Üí        -- and the world phase. No more onhitotherfn swapping.
   850‚Üí   850‚Üí        -- v2.0.51: Removed the day/dusk phase lock. The sentry is a STATIC

   859‚Üí   859‚Üí            local owner = GetSentryOwner(inst)

   865‚Üí   865‚Üí            inst._owner_shadow = shadow or false
