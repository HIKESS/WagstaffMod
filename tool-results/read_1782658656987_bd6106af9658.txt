    69→local function SpawnShadowClone(parent_buster)
    70→    if not parent_buster or not parent_buster:IsValid() then return nil end
    71→    
    72→    local x, y, z = parent_buster.Transform:GetWorldPosition()
    73→    
    74→    -- Spawn a shadow version using the SAME prefab type as parent (MK1, MK2 or MK3)
    75→    local parent_prefab = parent_buster.prefab
    76→    local clone_prefab = (parent_prefab == "williambuster3") and "williambuster3" or ((parent_prefab == "williambuster2") and "williambuster2" or "williambuster")
    77→    local clone = SpawnPrefab(clone_prefab)
    78→    if not clone then return nil end
    79→    
    80→    -- Position near the buster
    81→    local offset = 1.5
    82→    local angle = math.random() * 2 * math.pi
    83→    clone.Transform:SetPosition(x + math.cos(angle) * offset, 0, z + math.sin(angle) * offset)
    84→    
    85→    -- Make it look like a SHADOW version - TOTALLY BLACK (no color)
    86→    -- Apply immediately to prevent any visible color flash
    87→    local function ApplyCloneTint(c)
    88→        if c:IsValid() and c.AnimState then
    89→            c.AnimState:SetMultColour(0.01, 0.01, 0.01, 0.55)
    90→            c.AnimState:SetAddColour(0, 0, 0, 0)
    91→        end
    92→    end
    93→    ApplyCloneTint(clone)
    94→    
    95→    -- Keep enforcing black + semi-transparent color constantly to prevent any reset.
    96→    -- v2.0.57 FIX: clone was losing its dark tint when the player walked away.
    97→    -- Root cause: DoPeriodicTask is PAUSED while the entity is asleep (DST
    98→    -- dormancy — entities far from any player enter sleepstate and their
    99→    -- periodic tasks stop ticking). During dormancy the SG 'spawn'/'revived'
   100→    -- states (and some engine colour resets) could wipe the tint, and the
   101→    -- paused re-tint task couldn't correct it. Now we ALSO listen for
   102→    -- 'entitywake' (fires when the entity wakes back up as the player
   103→    -- approaches) and re-apply the tint immediately — closing the window
   104→    -- where the clone renders at full colour while far from the player.
   105→    clone:DoPeriodicTask(0.05, function() ApplyCloneTint(clone) end)
   106→    clone:ListenForEvent("entitywake", function(c) ApplyCloneTint(c) end)
   107→    -- Also re-apply on 'onremove'/'onpreload' is unnecessary — the clone is
   108→    -- transient (persists = false). But guard the SG 'spawn'/'revived' states
   109→    -- as well (already guarded in SGwilliambuster via shadow_buster_clone tag).
   110→    
   111→    -- Set up combat stats.
   112→    -- v2.0.39: clone damage 50% -> 35% of parent.
   113→    -- v2.0.57: clone damage 35% -> 40% of parent per user request (35% felt
   114→    -- too weak after the v2.0.50 revert stopped the clone hunting shadow
   115→    -- creatures — the clone now only mirrors the parent's target, so a bit
   116→    -- more damage keeps it a meaningful support fighter).
   117→    if clone.components.combat then
   118→        local parent_damage = parent_buster.components.combat and parent_buster.components.combat.defaultdamage or TUNING.WILLIAM_BUSTER_DAMAGE
   119→        clone.components.combat:SetDefaultDamage(parent_damage * 0.40)
   120→    end
   121→
   122→    -- v2.0.39: clone HP is now FINITE (50% of parent max HP) instead of
   123→    -- invincible. An immortal clone that also hunts shadows (v2.0.38) had no
   124→    -- trade-off. Now the clone can die if focused by enemies — the player must
   125→    -- position the parent well to keep the clone alive. Still absorbs damage
   126→    -- (50% absorption) so it's tanky but not unkillable.
   127→    if clone.components.health then
   128→        local parent_maxhp = parent_buster.components.health and parent_buster.components.health.maxhealth or TUNING.WILLIAM_BUSTER_HEALTH
   129→        clone.components.health:SetMaxHealth(parent_maxhp * 0.5)
   130→        clone.components.health:SetAbsorptionAmount(0.5)
   131→        -- Keep health regen so the clone recovers between fights (parent has regen too).
   132→        if clone.components.health.StartRegen then
   133→            clone.components.health:StartRegen(TUNING.WILLIAM_ROBOT_REGEN or 5, TUNING.WILLIAM_ROBOT_REGENPERIOD or 5)
   134→        end
   135→    end
   136→    
   137→    -- Don't consume fuel (it's a shadow manifestation)
   138→    if clone.components.fueled then
   139→        clone.components.fueled:StopConsuming()
   140→    end
   141→    
   142→    -- Tag as shadow clone.
   143→    -- v2.0.50: RE-ADDED "shadowcreature" tag. The clone IS a shadow, so
   144→    -- shadow creatures (Terrorbeak, Crawling Horror, Night Hands) are its
   145→    -- kin — they should NOT attack each other. This restores the
   146→    -- shadow-vs-shadow immunity: hostile shadow monsters treat the clone
   147→    -- as friendly and the clone does not hunt them. The clone now fights
   148→    -- ONLY the parent's target (normal combat), consistent with the
   149→    -- faction logic applied to the Brute taunt in v2.0.48 (each faction
   150→    -- doesn't fight its own kin). The v2.0.38 "hunts shadow creatures"
   151→    -- behavior is reverted as thematically wrong.
   152→    clone:AddTag("shadow_buster_clone")
   153→    clone:AddTag("shadowcreature")    -- v2.0.50: shadow-vs-shadow immunity
   154→    clone:AddTag("shadow_aligned")    -- v2.0.50: explicit rift faction tag
   155→    clone:AddTag("NOCLICK") -- Can't be clicked/interacted with
   156→    
   157→    -- CRITICAL: Do not save this clone - it should always be recreated on load
   158→    clone.persists = false
   159→    
   160→    -- Set up follower to follow parent Buster (needed for brain retarget)
   161→    if clone.components.follower and parent_buster:IsValid() then
   162→        clone.components.follower:SetLeader(parent_buster)
   163→    end
   164→    
   165→    -- Configure combat retarget: attack ONLY the parent's current target.
   166→    -- v2.0.50: REVERTED the v2.0.38 "hunts shadow creatures" behavior. The
   167→    -- clone is a shadow (shadowcreature + shadow_aligned tags), so it should
   168→    -- NOT hunt its own kin (Terrorbeak, Crawling Horror). It now mirrors the
   169→    -- parent's target only — a normal shadow-THEMED combat bot, not an
   170→    -- anti-shadow fighter. Consistent with the Brute taunt faction logic.
   171→    if clone.components.combat then
   172→        clone.components.combat:SetRetargetFunction(1, function(inst)
   173→            if not parent_buster:IsValid() then return nil end
   174→
   175→            -- Only target: parent's current target.
   176→            -- v2.0.41 FIX: guard with components.health check — IsDead() is a
   177→            -- health COMPONENT method, not an entity method.
   178→            if parent_buster.components.combat then
   179→                local parent_target = parent_buster.components.combat.target
   180→                if parent_target and parent_target:IsValid() and not parent_target:IsInLimbo()
   181→                    and parent_target.components.health ~= nil
   182→                    and not parent_target.components.health:IsDead() then
   183→                    -- v2.0.50: don't target shadow creatures (our kin).
   184→                    if not parent_target:HasTag("shadow") and not parent_target:HasTag("shadow_aligned") then
   185→                        return parent_target
   186→                    end
   187→                end
   188→            end
   189→
   190→            return nil
   191→        end)
   192→
   193→        clone.components.combat.canretarget = true
   194→        clone.components.combat:SetKeepTargetFunction(function(inst, target)
   195→            if not target or not target:IsValid() then return false end
   196→            if target:IsInLimbo() then return false end
   197→            if target.components.health == nil or target.components.health:IsDead() then return false end
   198→            -- v2.0.50: drop target if it's a shadow creature (our kin).
   199→            if target:HasTag("shadow") or target:HasTag("shadow_aligned") then return false end
   200→            -- Stay on target while parent is alive
   201→            if not parent_buster:IsValid() then return false end
   202→            return true
   203→        end)
   204→    end
   205→    
   206→    -- Remove name to avoid confusion
   207→    if clone.components.named then
   208→        clone.components.named:SetName("Shadow Buster")
   209→    end
   210→    
   211→    -- Monitor dusk state, parent health, and CLONE health.
   212→    -- v2.0.39: clone now has finite HP — also despawn when the clone itself dies
   213→    -- (instead of leaving a corpse that lingers until dusk ends).
   214→    clone:DoPeriodicTask(0.5, function()
   215→        if not parent_buster:IsValid() or not clone:IsValid() then
   216→            if clone:IsValid() then clone:Remove() end
   217→            return
   218→        end
   219→
   220→        -- v2.0.39: if the clone's HP ran out, despawn it with the shadow FX
   221→        -- (don't leave a dead clone body hanging around).
   222→        if clone.components.health and clone.components.health:IsDead() then
   223→            local remove_fx = SpawnPrefab("shadow_despawn")
   224→            if remove_fx then
   225→                local cx, cy, cz = clone.Transform:GetWorldPosition()
   226→                remove_fx.Transform:SetPosition(cx, cy, cz)
   227→                if remove_fx.SoundEmitter then
   228→                    remove_fx.SoundEmitter:KillAllSounds()
   229→                end
   230→            end
   231→            clone:Remove()
   232→            return
   233→        end
   234→
   235→        -- v2.0.58: Despawn when the parent's shadow pulse fades too — not only
   236→        -- when dusk ends. The pulse (parent_buster._aff_active) turns off when
   237→        -- the owner leaves combat or walks out of range, so the clone fades with
   238→        -- it, keeping pulse + clone in sync. (At dusk, _aff_active implies the
   239→        -- shadow path is active, so this check is safe.)
   240→        local parent_dead = not parent_buster.components.health or parent_buster.components.health:IsDead()
   241→        local pulse_faded = not TheWorld.state.isdusk or parent_buster._aff_active ~= true
   242→        if parent_dead or pulse_faded then
   243→            -- FX: shadow despawn (NO SOUND)
   244→            local remove_fx = SpawnPrefab("shadow_despawn")
   245→            if remove_fx then
   246→                local cx, cy, cz = clone.Transform:GetWorldPosition()
   247→                remove_fx.Transform:SetPosition(cx, cy, cz)
   248→                -- Kill all sounds on this FX
   249→                if remove_fx.SoundEmitter then
   250→                    remove_fx.SoundEmitter:KillAllSounds()
   251→                end
   252→            end
   253→            clone:Remove()
   254→            return
   255→        end
   256→        
   257→        -- Periodically suggest parent's target to the clone
   258→        if parent_buster.components.combat and parent_buster.components.combat.target then
   259→            local target = parent_buster.components.combat.target
   260→            if clone.components.combat and target:IsValid() and not target:IsInLimbo() then
   261→                clone.components.combat:SuggestTarget(target)
   262→            end
   263→        end
   264→    end)
   265→    
   266→    -- Visual effect on spawn (shadow materialization FX - shadows coming together)
   267→    local fx = SpawnPrefab("statue_transition_2")
   268→    if fx then
   269→        fx.Transform:SetPosition(clone.Transform:GetWorldPosition())
   270→    end
   271→    
   272→    -- Spawn sound
   273→    if clone.SoundEmitter then
   274→        clone.SoundEmitter:PlaySound("dontstarve/common/ghost_spawn")
   275→    end
   276→    
   277→    return clone

   280→local function LevelUp(inst, amount)
   281→        if inst.level < 3 and amount ~= nil then
   282→        inst.level = inst.level + amount
   283→        if inst.sg ~= nil then
   284→        inst.sg:GoToState("upgraded")

   290→                inst:AddTag("level"..inst.level)

   292→        inst.components.health:SetAbsorptionAmount(0+inst.level*0.05)
   293→        if inst.components.combat ~= nil then
   294→            -- v2.0.15 FIX: preserve tier bonus (MK2 +5, MK3 +10) instead of overriding it
   295→            local tier_bonus = 0

   298→            inst.components.combat:SetDefaultDamage(TUNING.WILLIAM_BUSTER_DAMAGE + tier_bonus + (inst.level*3))
   299→        end
   300→        end)

   329→local function OnAddFuel(inst, fuelvalue, fuelitem)
   330→    -- v2.0.70 FIX: DST's `fueled` component calls ontakefuelfn as
   331→    -- (inst, fuelvalue, ...) — NOT (inst, doer, fuelitem). Previous code
   332→    -- treated the fuelvalue number as `doer`, crashing with "attempt to
   333→    -- index local 'doer' (a number value)" when fuel was already full.
   334→    -- v2.0.63: reject fuel when already full (vanilla-style feedback).
   335→    if inst.components.fueled and inst.components.fueled:IsFull() then
   336→        local player = FindClosestPlayerToInst(inst, 10, true)
   337→        if player and player.components.talker then
   338→            player.components.talker:Say("It's already full!")
   339→        end
   340→        return false
   341→    end
   342→    -- Only accept generator fuel (gears, etc.), reject common fuel like wood.
   343→    -- NOTE: in this DST version, fuelitem is nil from ontakefuelfn (DST passes
   344→    -- only fuelvalue as the 2nd arg), so this validation is currently a no-op.
   345→    -- Fuel-type filtering should be done via the fueled component's
   346→    -- SetFuelAcceptingTest or similar, not here. Left in place for safety.
   347→    if fuelitem and fuelitem.prefab then
   348→        if not VALID_BUSTER_FUELS[fuelitem.prefab] then
   349→            -- Not a valid fuel for Buster Bot
   350→            local player = FindClosestPlayerToInst(inst, 10, true)

   353→            end
