    82‚Üí    82‚Üílocal function weighted_random_choice(items)
    83‚Üí    83‚Üí    local total = 0
    84‚Üí    84‚Üí    for _, v in ipairs(items) do
    85‚Üí    85‚Üí        total = total + v.weight
    86‚Üí    86‚Üí    end
    87‚Üí    87‚Üí    local rand = math.random() * total
    88‚Üí    88‚Üí    for _, v in ipairs(items) do
    89‚Üí    89‚Üí        rand = rand - v.weight
    90‚Üí    90‚Üí        if rand <= 0 then
    91‚Üí    91‚Üí            return v.item
    92‚Üí    92‚Üí        end
    93‚Üí    93‚Üí    end
    94‚Üí    94‚Üí    return items[1].item
    95‚Üí    95‚Üíend

   215‚Üí   215‚Üí            local function GetBuilder(inst)
