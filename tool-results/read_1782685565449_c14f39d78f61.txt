    74→local function GetOwner(inst)
    75→    -- Try follower component first (for bots that still use it, e.g. buster/brute)
    76→    if inst.components.follower then
    77→        local leader = inst.components.follower:GetLeader()
    78→        if leader then return leader end
    79→    end
    80→    -- Fallback: resolve owner by stored GUID (Ballistic MK3 turret mode)
    81→    if inst._owner_guid then
    82→        local owner = Ents[inst._owner_guid]
    83→        if owner and owner:IsValid() then return owner end
    84→    end
    85→    return nil
    86→end

   110→-- Passive Lantern Light for MK3 Ballistic Bot
   111→-- Two separate lights: one fixed (small, steady, yellow) and one pulse (large, periodic, yellow)
   112→-- Both only active at night (auto-toggle)
   113→
   114→-- Lightning yellow colour
   115→local LIGHT_R, LIGHT_G, LIGHT_B = 1, 1, 0.3
   116→
   117→-- Fixed light: small steady glow, always on at night
   118→local FIX_RADIUS = 1.5
   119→local FIX_INTENSITY = 0.8
   120→local FIX_FALLOFF = 0.5
   121→
   122→-- Pulse light: large flash, ~same size as a max campfire
   123→local PULSE_RADIUS = 5.5
   124→local PULSE_INTENSITY = 0.85
   125→local PULSE_FALLOFF = 0.5
   126→local PULSE_INTERVAL = 1.5    -- seconds between pulses
   127→local PULSE_DURATION = 0.35   -- how long each pulse flash lasts

   129→-- Registered prefab for the MK3 pulse light (DST requires registered prefabs
   130→-- for network replication; CreateEntity() without a prefab crashes with
   131→-- "AllocReplica Invalid Prefab").
   132→local function ballistic_pulse_light_fn()
   133→    local inst = CreateEntity()
   134→    inst.entity:AddTransform()
   135→    inst.entity:AddNetwork()
   136→    inst.entity:AddLight()
   137→    inst.Light:Enable(false)
   138→    inst.Light:SetRadius(0)
   139→    inst.Light:SetIntensity(PULSE_INTENSITY)
   140→    inst.Light:SetFalloff(PULSE_FALLOFF)
   141→    inst.Light:SetColour(LIGHT_R, LIGHT_G, LIGHT_B)
   142→    inst.persists = false
   143→    return inst
   144→end

   146→local function StartLightOrb(inst)
   147→    if inst._lightorb_active then return end
   148→    if inst.components.fueled:IsEmpty() then return end
   149→    if not TheWorld.state.isnight then return end
   150→    inst._lightorb_active = true
   151→    inst:AddTag("lantern")
   152→
   153→    -- FIXED LIGHT: set once, never changes while active
   154→    inst.Light:SetRadius(FIX_RADIUS)
   155→    inst.Light:SetIntensity(FIX_INTENSITY)
   156→    inst.Light:SetFalloff(FIX_FALLOFF)
   157→    inst.Light:SetColour(LIGHT_R, LIGHT_G, LIGHT_B)
   158→    inst.Light:Enable(true)
   159→
   160→    -- PULSE LIGHT: re-spawn if missing (e.g. after load or if removed)
   161→    if not inst._pulse_light or not inst._pulse_light:IsValid() then
   162→        inst._pulse_light = SpawnPrefab("ballistic_pulse_light")
   163→        if inst._pulse_light then
   164→            inst._pulse_light.entity:SetParent(inst.entity)
   165→            inst._pulse_light.Transform:SetPosition(0, 1, 0)
   166→        else
   167→            return  -- Cannot spawn pulse light
   168→        end
   169→    end
   170→
   171→    -- Periodic pulse: flash big, then dim, repeat
   172→    inst._lightorb_tick = inst:DoPeriodicTask(PULSE_INTERVAL, function()
   173→        if not inst._lightorb_active then return end
   174→        if inst.components.fueled:IsEmpty() then
   175→            StopLightOrb(inst)
   176→            return
   177→        end
   178→
   179→        -- Flash the pulse light ON
   180→        if inst._pulse_light and inst._pulse_light:IsValid() then
   181→            inst._pulse_light.Light:SetRadius(PULSE_RADIUS)
   182→            inst._pulse_light.Light:Enable(true)
   183→        end
   184→
   185→        -- Spawn spark FX
   186→        if inst._lightorb_fx and inst._lightorb_fx:IsValid() then
   187→            inst._lightorb_fx:Remove()
   188→        end
   189→        local newfx = SpawnPrefab("wx78_big_spark")
   190→        if newfx then
   191→            newfx.entity:SetParent(inst.entity)
   192→            newfx.Transform:SetPosition(0, 1, 0)
   193→            inst._lightorb_fx = newfx
   194→        end
   195→
   196→        -- Turn pulse OFF after short duration
   197→        inst._pulse_dim_task = inst:DoTaskInTime(PULSE_DURATION, function()
   198→            if inst._pulse_light and inst._pulse_light:IsValid() then
   199→                inst._pulse_light.Light:SetRadius(0)
   200→                inst._pulse_light.Light:Enable(false)
   201→            end
   202→            if inst._lightorb_fx and inst._lightorb_fx:IsValid() then
   203→                inst._lightorb_fx:Remove()
   204→                inst._lightorb_fx = nil
   205→            end
   206→        end)
   207→    end)
   208→end

   210→local function StopLightOrb(inst)
   211→    inst._lightorb_active = false
   212→
   213→    -- Remove pulse light entity (spawned via SpawnPrefab, so safe to Remove)
   214→    if inst._pulse_dim_task then
   215→        inst._pulse_dim_task:Cancel()
   216→        inst._pulse_dim_task = nil
   217→    end
   218→    if inst._pulse_light and inst._pulse_light:IsValid() then
   219→        inst._pulse_light:Remove()
   220→        inst._pulse_light = nil
   221→    end
   222→
   223→    -- Remove FX
   224→    if inst._lightorb_fx and inst._lightorb_fx:IsValid() then
   225→        inst._lightorb_fx:Remove()
   226→    end
   227→    inst._lightorb_fx = nil
   228→
   229→    -- Disable fixed light
   230→    if inst.Light then
   231→        inst.Light:Enable(false)
   232→    end
   233→    inst:RemoveTag("lantern")
   234→
   235→    -- Stop tick
   236→    if inst._lightorb_tick ~= nil then
   237→        inst._lightorb_tick:Cancel()
   238→        inst._lightorb_tick = nil
   239→    end
   240→end

   242→local function maketurret(inst, pt, charge)
   243→    local is_mk3 = inst.was_mk3 or (inst.upgradelevel_mk3 ~= nil and inst.upgradelevel_mk3 >= 150)
   244→    local is_mk2 = is_mk3 or inst.was_mk2 or (inst.upgradelevel ~= nil and inst.upgradelevel >= 100)
   245→    local prefab = is_mk3 and "williamballistic3" or (is_mk2 and "williamballistic2" or "williamballistic")
   246→    local bot = SpawnPrefab(prefab)
   247→    if bot ~= nil then
   248→        bot.Physics:SetCollides(false)
   249→        bot.Physics:Teleport(pt.x, 0, pt.z)
   250→        bot.Physics:SetCollides(true)
   251→        bot.sg:GoToState("revived", charge)
   252→        bot.components.health:SetCurrentHealth(inst.components.health.currenthealth)
   253→        bot.components.fueled.currentfuel = inst.components.fueled.currentfuel
   254→        bot.level = inst.level
   255→        bot.upgradelevel = inst.upgradelevel or 0
   256→        if is_mk2 then
   257→            bot.upgradelevel_mk3 = inst.upgradelevel_mk3 or 0
   258→        end
   259→        bot:PushEvent("levelup")
   260→        inst:Remove()
   261→    end
   262→end

   298→local function retargetfn(inst)
   299→    local playertargets = {}
   300→    for i, v in ipairs(AllPlayers) do
   301→        if v.components.combat.target ~= nil then
   302→            playertargets[v.components.combat.target] = true
   303→        end
   304→    end
   305→
   306→    return FindEntity(inst, PLACER_SCALE*10,
   307→        function(guy)
   308→            if not inst.components.combat:CanTarget(guy) then return false end
   309→            -- Target enemies already fighting the player/minions
   310→            if playertargets[guy] then return true end
   311→            -- Also target any hostile creature (monster) in range
   312→            if guy:HasTag("monster") and not guy:HasTag("player") and not guy:HasTag("companion") then
   313→                return true
   314→            end
   315→            -- Target creatures attacking players or willminion
   316→            if guy.components.combat.target ~= nil
   317→                and (guy.components.combat.target:HasTag("player") or guy.components.combat.target:HasTag("willminion")) then
   318→                return true
   319→            end
   320→            return false
   321→        end,
   322→        { "_combat" }, --see entityreplica.lua
   323→        { "INLIMBO", "player" }
   324→    )
   325→end
