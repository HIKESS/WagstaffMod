   131→   131→local function StartLightOrb(inst)
   132→   132→    if inst._lightorb_active then return end
   133→   133→    if inst.components.fueled:IsEmpty() then return end
   134→   134→    if not TheWorld.state.isnight then return end
   135→   135→    inst._lightorb_active = true
   136→   136→    inst:AddTag("lantern")
   137→   137→
   138→   138→    -- FIXED LIGHT: set once, never changes while active
   139→   139→    inst.Light:SetRadius(FIX_RADIUS)
   140→   140→    inst.Light:SetIntensity(FIX_INTENSITY)
   141→   141→    inst.Light:SetFalloff(FIX_FALLOFF)
   142→   142→    inst.Light:SetColour(LIGHT_R, LIGHT_G, LIGHT_B)
   143→   143→    inst.Light:Enable(true)
   144→   144→
   145→   145→    -- PULSE LIGHT: re-spawn if missing (e.g. after load or if removed)
   146→   146→    if not inst._pulse_light or not inst._pulse_light:IsValid() then
   147→   147→        inst._pulse_light = SpawnPrefab("ballistic_pulse_light")
   148→   148→        if inst._pulse_light then
   149→   149→            inst._pulse_light.entity:SetParent(inst.entity)
   150→   150→            inst._pulse_light.Transform:SetPosition(0, 1, 0)
   151→   151→        else
   152→   152→            return  -- Cannot spawn pulse light
   153→   153→        end
   154→   154→    end
   155→   155→
   156→   156→    -- Periodic pulse: flash big, then dim, repeat
   157→   157→    inst._lightorb_tick = inst:DoPeriodicTask(PULSE_INTERVAL, function()
   158→   158→        if not inst._lightorb_active then return end
   159→   159→        if inst.components.fueled:IsEmpty() then
   160→   160→            StopLightOrb(inst)
   161→   161→            return
   162→   162→        end
   163→   163→
   164→   164→        -- Flash the pulse light ON
   165→   165→        if inst._pulse_light and inst._pulse_light:IsValid() then
   166→   166→            inst._pulse_light.Light:SetRadius(PULSE_RADIUS)
   167→   167→            inst._pulse_light.Light:Enable(true)
   168→   168→        end
   169→   169→
   170→   170→        -- Spawn spark FX
   171→   171→        if inst._lightorb_fx and inst._lightorb_fx:IsValid() then
   172→   172→            inst._lightorb_fx:Remove()
   173→   173→        end
   174→   174→        local newfx = SpawnPrefab("wx78_big_spark")
   175→   175→        if newfx then
   176→   176→            newfx.entity:SetParent(inst.entity)
   177→   177→            newfx.Transform:SetPosition(0, 1, 0)
   178→   178→            inst._lightorb_fx = newfx
   179→   179→        end
   180→   180→
   181→   181→        -- Turn pulse OFF after short duration
   182→   182→        inst._pulse_dim_task = inst:DoTaskInTime(PULSE_DURATION, function()
   183→   183→            if inst._pulse_light and inst._pulse_light:IsValid() then
   184→   184→                inst._pulse_light.Light:SetRadius(0)
   185→   185→                inst._pulse_light.Light:Enable(false)
   186→   186→            end
   187→   187→            if inst._lightorb_fx and inst._lightorb_fx:IsValid() then
   188→   188→                inst._lightorb_fx:Remove()
   189→   189→                inst._lightorb_fx = nil
   190→   190→            end
   191→   191→        end)
   192→   192→    end)
   193→   193→end
