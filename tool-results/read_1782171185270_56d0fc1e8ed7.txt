    63‚Üí    63‚Üílocal function weighted_random_choice(items)
    64‚Üí    64‚Üí    local total = 0
    65‚Üí    65‚Üí    for _, v in ipairs(items) do
    66‚Üí    66‚Üí        total = total + v.weight
    67‚Üí    67‚Üí    end
    68‚Üí    68‚Üí    local rand = math.random() * total
    69‚Üí    69‚Üí    for _, v in ipairs(items) do
    70‚Üí    70‚Üí        rand = rand - v.weight
    71‚Üí    71‚Üí        if rand <= 0 then
    72‚Üí    72‚Üí            return v.item
    73‚Üí    73‚Üí        end
    74‚Üí    74‚Üí    end
    75‚Üí    75‚Üí    return items[1].item
    76‚Üí    76‚Üíend
