    63→local function SpawnShadowClone(parent_buster)
    64→    if not parent_buster or not parent_buster:IsValid() then return nil end
    65→    
    66→    local x, y, z = parent_buster.Transform:GetWorldPosition()
    67→    
    68→    -- Spawn a shadow version using the SAME prefab type as parent (MK1, MK2 or MK3)
    69→    local parent_prefab = parent_buster.prefab
    70→    local clone_prefab = (parent_prefab == "williambuster3") and "williambuster3" or ((parent_prefab == "williambuster2") and "williambuster2" or "williambuster")
    71→    local clone = SpawnPrefab(clone_prefab)
    72→    if not clone then return nil end
    73→    
    74→    -- Position near the buster
    75→    local offset = 1.5
    76→    local angle = math.random() * 2 * math.pi
    77→    clone.Transform:SetPosition(x + math.cos(angle) * offset, 0, z + math.sin(angle) * offset)
    78→    
    79→    -- Make it look like a SHADOW version - TOTALLY BLACK (no color)
    80→    -- Apply immediately to prevent any visible color flash
    81→    if clone.AnimState then
    82→        clone.AnimState:SetMultColour(0.01, 0.01, 0.01, 0.55)
    83→        clone.AnimState:SetAddColour(0, 0, 0, 0)
    84→    end
    85→    
    86→    -- Keep enforcing black + semi-transparent color constantly to prevent any reset
    87→    clone:DoPeriodicTask(0.05, function()
    88→        if clone:IsValid() and clone.AnimState then
    89→            clone.AnimState:SetMultColour(0.01, 0.01, 0.01, 0.55)
    90→            clone.AnimState:SetAddColour(0, 0, 0, 0)
    91→        end
    92→    end)
    93→    
    94→    -- Set up combat stats (50% of parent)
    95→    if clone.components.combat then
    96→        local parent_damage = parent_buster.components.combat and parent_buster.components.combat.defaultdamage or TUNING.WILLIAM_BUSTER_DAMAGE
    97→        clone.components.combat:SetDefaultDamage(parent_damage * 0.5)
    98→    end
    99→    
   100→    -- Make health invincible (shadow clone takes no damage)
   101→    if clone.components.health then
   102→        clone.components.health:SetInvincible(true)
   103→        clone.components.health:SetAbsorptionAmount(1)
   104→    end
   105→    
   106→    -- Don't consume fuel (it's a shadow manifestation)
   107→    if clone.components.fueled then
   108→        clone.components.fueled:StopConsuming()
   109→    end
   110→    
   111→    -- Tag as shadow clone
   112→    clone:AddTag("shadow_buster_clone")
   113→    clone:AddTag("shadowcreature")
   114→    clone:AddTag("NOCLICK") -- Can't be clicked/interacted with
   115→    
   116→    -- CRITICAL: Do not save this clone - it should always be recreated on load
   117→    clone.persists = false
   118→    
   119→    -- Set up follower to follow parent Buster (needed for brain retarget)
   120→    if clone.components.follower and parent_buster:IsValid() then
   121→        clone.components.follower:SetLeader(parent_buster)
   122→    end
   123→    
   124→    -- Configure combat retarget to attack parent's target
   125→    if clone.components.combat then
   126→        -- Clear default retarget and set new one
   127→        clone.components.combat:SetRetargetFunction(1, function(inst)
   128→            -- Check parent_buster target
   129→            if parent_buster:IsValid() and parent_buster.components.combat then
   130→                local parent_target = parent_buster.components.combat.target
   131→                if parent_target and parent_target:IsValid() and not parent_target:IsInLimbo() then
   132→                    return parent_target
   133→                end
   134→            end
   135→            return nil
   136→        end)
   137→    end
   138→    
   139→    -- Remove name to avoid confusion
   140→    if clone.components.named then
   141→        clone.components.named:SetName("Shadow Buster")
   142→    end
   143→    
   144→    -- Monitor dusk state and parent health - REMOVE when dusk ends or parent dies
   145→    clone:DoPeriodicTask(0.5, function()
   146→        if not parent_buster:IsValid() or not clone:IsValid() then
   147→            if clone:IsValid() then clone:Remove() end
   148→            return
   149→        end
   150→        
   151→        -- Check if dusk ended (not dusk anymore) or parent is dead
   152→        if not TheWorld.state.isdusk or not parent_buster.components.health or parent_buster.components.health:IsDead() then
   153→            -- FX: shadow despawn (NO SOUND)
   154→            local remove_fx = SpawnPrefab("shadow_despawn")
   155→            if remove_fx then
   156→                local cx, cy, cz = clone.Transform:GetWorldPosition()
   157→                remove_fx.Transform:SetPosition(cx, cy, cz)
   158→                -- Kill all sounds on this FX
   159→                if remove_fx.SoundEmitter then
   160→                    remove_fx.SoundEmitter:KillAllSounds()
   161→                end
   162→            end
   163→            clone:Remove()
   164→            return
   165→        end
   166→        
   167→        -- Periodically suggest parent's target to the clone
   168→        if parent_buster.components.combat and parent_buster.components.combat.target then
   169→            local target = parent_buster.components.combat.target
   170→            if clone.components.combat and target:IsValid() and not target:IsInLimbo() then
   171→                clone.components.combat:SuggestTarget(target)
   172→            end
   173→        end
   174→    end)
   175→    
   176→    -- Visual effect on spawn (shadow materialization FX - shadows coming together)
   177→    local fx = SpawnPrefab("statue_transition_2")
   178→    if fx then
   179→        fx.Transform:SetPosition(clone.Transform:GetWorldPosition())
   180→    end
   181→    
   182→    -- Spawn sound
   183→    if clone.SoundEmitter then
   184→        clone.SoundEmitter:PlaySound("dontstarve/common/ghost_spawn")
   185→    end
   186→    
   187→    return clone
   188→end
