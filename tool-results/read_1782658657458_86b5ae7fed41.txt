    97→local function TauntCreatures(inst)
    98→    if not inst.components.health:IsDead() then
    99→        local x, y, z = inst.Transform:GetWorldPosition()
   100→
   101→        -- v2.0.48: Affinity-aware taunt for MK3 brutes. Each affinity now tanks
   102→        -- its ENEMY faction (mirrors the sentry faction logic — lunar and shadow
   103→        -- are opposing rift factions). Previously the taunt was affinity-blind:
   104→        -- the shadow Brute would draw aggro from shadow-aligned creatures (its
   105→        -- own kin), which was thematically wrong.
   106→        --   Celestial (day) -> taunts shadow-aligned + neutral hostiles,
   107→        --                       EXCLUDES lunar-aligned (allies).
   108→        --   Shadow (dusk)   -> taunts lunar-aligned + neutral hostiles,
   109→        --                       EXCLUDES shadow-aligned (allies).
   110→        --   Base / MK1 / MK2 / affinity inactive -> original behavior
   111→        --                       (exclude the legacy "shadow" tag so basic
   112→        --                        brutes don't pull nightmare creatures).
   113→        local is_mk3 = inst.prefab == "williambrute3"
   114→        local celestial_active = is_mk3 and TheWorld.state.isday and OwnerHasCelestial(inst)
   115→        local shadow_active    = is_mk3 and TheWorld.state.isdusk and OwnerHasShadow(inst)
   116→
   117→        local cant_tags, affinity_filter
   118→        if celestial_active then
   119→            -- Celestial Brute tanks the shadow faction (its enemy).
   120→            cant_tags = { "INLIMBO", "player", "companion", "epic", "notaunt" }
   121→            affinity_filter = function(v) return not v:HasTag("lunar_aligned") end
   122→        elseif shadow_active then
   123→            -- Shadow Brute tanks the lunar faction (its enemy).
   124→            cant_tags = { "INLIMBO", "player", "companion", "epic", "notaunt" }
   125→            affinity_filter = function(v) return not v:HasTag("shadow_aligned") end
   126→        else
   127→            -- Base/MK1/MK2 or affinity inactive: original behavior.
   128→            cant_tags = { "INLIMBO", "player", "companion", "epic", "notaunt", "shadow" }
   129→        end
   130→
   131→        for i, v in ipairs(TheSim:FindEntities(x, y, z, 7, { "_combat", "locomotor" }, cant_tags)) do
   132→            if IsTauntable(inst, v) and (affinity_filter == nil or affinity_filter(v)) then

   135→        end

   216→-- v2.0.71: distance from leader beyond which the brute disengages from combat.
   217→-- Mirrors KEEP_WORKING_DIST in the brute brain. When the player runs farther
   218→-- than this, the brute drops its target and retreats to the leader instead of
   219→-- fighting to the death.
   220→local BRUTE_DISENGAGE_DIST = 14
   221→-- v2.0.71: HP fraction at or below which the brute retreats (it used to fight
   222→-- to the death). Tuned for a tank: MK1 1500 HP -> retreat at 375, MK2 2100 ->
   223→-- 525, MK3 2600 -> 650.
   224→local BRUTE_FLEE_HP_PCT = 0.25

   226→local function keeptargetfn(inst, target)
   227→    --give up on dead guys, or guys in the dark, or werepigs
   228→    if not inst.components.combat:CanTarget(target) then
   229→        return false
   230→    end
   231→    -- v2.0.71: disengage when the leader has run too far away. Without this the
   232→    -- brute keeps fighting (and re-acquiring herd members via OnAttacked) even
   233→    -- after the player flees — "tentei correr, não consegui". Dropping the
   234→    -- target at the component level means ChaseAndAttack in the brain has
   235→    -- nothing to chase, and retargetfn won't find new targets once the player
   236→    -- is no longer in combat.
   237→    if inst.components.follower ~= nil and inst.components.follower.leader ~= nil then
   238→        if not inst:IsNear(inst.components.follower.leader, BRUTE_DISENGAGE_DIST) then
   239→            return false
   240→        end
   241→    end
   242→    -- v2.0.71: retreat when critically low on HP so the brute doesn't fight to
   243→    -- the death. The brain's LowHP Retreat node then drives it to the leader
   244→    -- (or away from threats if it has no leader).
   245→    if inst.components.health ~= nil and inst.components.health.maxhealth > 0 then
   246→        local pct = inst.components.health.currenthealth / inst.components.health.maxhealth
   247→        if pct <= BRUTE_FLEE_HP_PCT then
   248→            return false
   249→        end
   250→    end
   251→    return true
   252→end

   279→local function TurnOff(inst, doer, instant)
   280→    local GLOBAL = _G
   281→    inst.on = false
   282→
   283→    -- v2.0.69: Drop the follower leader when deactivated.
   284→    -- The DST `follower` component has a built-in catch-up teleport that fires
   285→    -- at the COMPONENT level (in OnUpdate), bypassing the brain entirely. Even
   286→    -- though the brain's `Active` WhileNode gates the Follow behaviour when
   287→    -- inst.on == false, the follower component still has a leader reference and
   288→    -- keeps teleporting the deactivated bot to the leader. Butler/buster avoid
   289→    -- this because their `empty` husk is a fresh entity with no leader; the
   290→    -- brute reuses the same entity, so we must explicitly drop the leader here.
   291→    -- TurnOn re-acquires the leader from the activating player (or nearest
   292→    -- player) on reactivation, and william_acts.lua allows WILLYRAISE when the
   293→    -- leader is nil so the bot can be reactivated after deactivation.
   294→    if inst.components.follower ~= nil then
   295→        inst.components.follower:StopFollowing()
   296→    end
   297→
   298→                if inst._task ~= nil then
   299→                    inst._task:Cancel()
   300→            inst._task = nil
   301→        end
   302→            MakeHauntableWork(inst)
   303→        inst:RemoveTag("scarytoprey")
   304→        inst:RemoveTag("alive")
   305→        inst:RemoveTag("ebuild_wrenchable")  -- Prevent wrench interaction when OFF
   306→        inst:AddTag("notarget")
   307→
   308→    -- v2.0.81 FIX: Do NOT RemoveComponent("container") on deactivation.
   309→    -- That destroys the items stored inside (DST drops/destroys them when
   310→    -- the component is removed), so reactivating the bot gave back an empty
   311→    -- chest and "itens do chest somen quando ele é desativado/reativado".
   312→    -- Instead, just remove the `container` tag so the chest can't be opened
   313→    -- while the bot is off. The component (and its items) persist through
   314→    -- deactivate/reactivate and through save/load (DST's native container
   315→    -- OnSave/OnLoad handles the items now that the component always exists).
   316→    -- Server-only: the component Close() and tag are server-authoritative.
   317→    if not GLOBAL.TheWorld.ismastersim then return end
   318→
   319→    if inst:HasTag("container") then
   320→        inst:RemoveTag("container")
   321→        if inst.components.container then
   322→            pcall(function() inst.components.container:Close() end)
   323→        end
   324→    end
   325→
   326→    inst.components.fueled:StopConsuming()
   327→        inst.components.combat:SetTarget(nil)
   328→    inst.components.combat:SetRetargetFunction(nil)
   329→    inst.components.combat:SetKeepTargetFunction(nil)
   330→    inst.sg:GoToState("turn_off")
   331→    
   332→    -- When OFF with no fuel: hammering BREAKS the bot (like buster/ballistic behavior)
   333→    -- When OFF with fuel: no workable — WILLYRAISE action handles reactivation
   334→    if inst.components.fueled:IsEmpty() then
   335→        if inst.components.workable == nil then
   336→            inst:AddComponent("workable")
   337→        end
   338→        inst.components.workable:SetWorkAction(ACTIONS.HAMMER)
   339→        inst.components.workable:SetWorkLeft(4)
   340→        inst.components.workable:SetOnFinishCallback(OnHammered)
   341→        inst.components.workable:SetOnWorkCallback(onworked)
   342→        -- Add fuel listener: when fuel is added while OFF, remove HAMMER so WILLYRAISE can activate
   343→        inst._fuel_activate_listener = inst:ListenForEvent("percentusedchange", function()
   344→            if inst.on == false and not inst.components.fueled:IsEmpty() then
   345→                if inst.components.workable then
   346→                    inst:RemoveComponent("workable")
   347→                end
   348→                -- WILLYRAISE action will now handle reactivation
   349→                if inst._fuel_activate_listener then
   350→                    inst:RemoveEventCallback("percentusedchange", inst._fuel_activate_listener)
   351→                    inst._fuel_activate_listener = nil
   352→                end
   353→            end
   354→        end)
   355→    else
   356→        -- Has fuel: remove workable so WILLYRAISE "Activate" action is available
   357→        if inst.components.workable then
   358→            inst:RemoveComponent("workable")
   359→        end
   360→    end
   361→end

   510→local function onload(inst, data)
   511→    if data == nil then return end
   512→
   513→    inst.on = data.on
   514→    if data.level ~= nil then
   515→        inst.level = data.level
   516→        if inst.level > 0 then inst:DoTaskInTime(0, LevelUp) end
   517→    end
   518→
   519→    -- Restore upgradelevel
   520→    if data.upgradelevel ~= nil then
   521→        inst.upgradelevel = data.upgradelevel
   522→    end
   523→
   524→    -- Restore leader (MK1 now has follower too)
   525→    if data.leader_guid ~= nil and inst.components.follower ~= nil then
   526→        inst:DoTaskInTime(0, function()
